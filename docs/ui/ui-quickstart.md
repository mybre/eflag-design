## 在项目中使用

### 安装

```bash
$ npm i @eflag/ui --save
```

### 示例

```jsx | pure
import { Lottie } from '@eflag/ui';

const App = () => {
  return (
    <Lottie
      path="https://assets9.lottiefiles.com/packages/lf20_WPqksadnNs.json"
      style={{
        height: 200,
      }}
    />
  );
};
```
